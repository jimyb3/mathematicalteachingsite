<script type="text/javascript">
	$(document).ready(function prog(){
	for(var i=1;i<=3){
		var grade=document.getElementById("progressbar"+i).value;
		document.getElementById("progressbar"+i).value=0;
		
		var goMore=function(){
			document.getElementById("progressbar"+i).value=document.getElementById("progressbar"+i).value+1;
						
			if(document.getElementById("progressbar"+i).value==grade){
				clearInterval(animate);
				i=i+1;		
			}
		};
		
		var animate=setInterval(function() {
		    goMore();}, 100);
	}
	});</script>
